def calculer_distance_vol_oiseau(latA : float, lngA : float, latB : float, lngB : float) :
    '''
    Calcule la distance √† vol d'oiseau entre deux points A(latA, lngA) et B(latB, lngB) du globe √† partir de leurs coordonn√©es GPS.
    '''

    distance = 0
